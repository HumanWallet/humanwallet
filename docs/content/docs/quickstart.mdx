# Quick Start Guide

This guide will walk you through integrating the HumanWallet connector into your React application, step by step. We'll build a complete integration that enables biometric authentication and account abstraction for your users.

## Prerequisites

Before we begin, make sure you have:

- **Node.js 18+** installed
- A **ZeroDev Project ID** (get one at [dashboard.zerodev.app](https://dashboard.zerodev.app/))
- Basic knowledge of **React** and **TypeScript**
- Familiarity with **Wagmi** (optional but helpful)

## Step 1: Create Your Project

First, create a new React project with Vite:

```bash
npm create vite@latest my-humanwallet-app -- --template react-ts
cd my-humanwallet-app
npm install
```

## Step 2: Install Dependencies

Install the required packages for HumanWallet integration:

```bash
# Core dependencies
npm install @humanwallet/connector viem wagmi

# React Query for caching
npm install @tanstack/react-query

# Optional: UI library for better components
npm install @radix-ui/react-dialog @radix-ui/react-alert-dialog
npm install lucide-react class-variance-authority clsx

# Development dependencies
npm install --save-dev vite-plugin-node-polyfills
```

## Step 5: Create Wagmi Configuration

Create `src/wagmi/config.ts`:

```typescript
// src/wagmi/config.ts
import { createConfig, http } from 'wagmi'
import { mainnet, sepolia, polygon } from 'wagmi/chains'
import { humanWalletConnector } from '@humanwallet/connector'

// Get project ID from environment variables
const ZERODEV_PROJECT_ID = import.meta.env.VITE_ZERODEV_PROJECT_ID

export const config = createConfig({
  chains: [sepolia, mainnet, polygon],
  connectors: [
    humanWalletConnector({
      projectId: ZERODEV_PROJECT_ID,
      appName: "My HumanWallet App",
      passkeyName: "My Wallet",
      dappMetadata: {
        name: "My DApp",
        url: "https://mydapp.com",
        iconUrl: "https://mydapp.com/icon.png",
        description: "My awesome DApp"
      },
      logging: {
        developerMode: true // Enable for development
      }
    }),
  ],
  transports: {
    [sepolia.id]: http(),
    [mainnet.id]: http(),
    [polygon.id]: http(),
  },
})

// TypeScript module augmentation for better type safety
declare module 'wagmi' {
  interface Register {
    config: typeof config
  }
}
```

## Step 6: Create Wagmi Provider

Create `src/wagmi/provider.tsx`:

```tsx
// src/wagmi/provider.tsx
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { WagmiProvider } from 'wagmi'
import { config } from './config'

const queryClient = new QueryClient()

export function WagmiProviders({ children }: { children: React.ReactNode }) {
  return (
    <WagmiProvider config={config}>
      <QueryClientProvider client={queryClient}>
        {children}
      </QueryClientProvider>
    </WagmiProvider>
  )
}
```

## Step 8: Create Connection Component

Create `src/components/ConnectWallet.tsx`:

```tsx
// src/components/ConnectWallet.tsx
import { useConnect, useAccount, useDisconnect } from 'wagmi'
import { useState } from 'react'

export function ConnectWallet() {
  const { connectors, connect, isPending, error } = useConnect()
  const { address, isConnected } = useAccount()
  const { disconnect } = useDisconnect()
  const [isConnecting, setIsConnecting] = useState(false)

  const humanWallet = connectors.find(c => c.id === 'humanWallet')

  const handleConnect = async () => {
    if (!humanWallet) return
    setIsConnecting(true)
    try {
      await connect({ connector: humanWallet })
    } finally {
      setIsConnecting(false)
    }
  }

  const handleCreateNew = async () => {
    if (!humanWallet) return
    setIsConnecting(true)
    try {
      // Force create new wallet
      await (connect as any)({ 
        connector: humanWallet, 
        forceCreate: true 
      })
    } finally {
      setIsConnecting(false)
    }
  }

  if (isConnected && address) {
    return (
      <div className="p-6 border rounded-lg">
        <h3 className="text-lg font-semibold mb-2">‚úÖ Connected</h3>
        <p className="text-sm text-gray-600 mb-4">
          Address: {address.slice(0, 6)}...{address.slice(-4)}
        </p>
        <button
          onClick={() => disconnect()}
          className="px-4 py-2 bg-red-500 text-white rounded hover:bg-red-600"
        >
          Disconnect
        </button>
      </div>
    )
  }

  return (
    <div className="p-6 border rounded-lg">
      <h3 className="text-lg font-semibold mb-4">üîê Connect HumanWallet</h3>
      
      {error && (
        <div className="mb-4 p-3 bg-red-50 border border-red-200 rounded">
          <p className="text-red-700 text-sm">{error.message}</p>
        </div>
      )}

      <div className="space-y-3">
        <button
          onClick={handleConnect}
          disabled={isConnecting || isPending}
          className="w-full px-4 py-3 bg-blue-500 text-white rounded hover:bg-blue-600 disabled:opacity-50"
        >
          {isConnecting || isPending ? 'Connecting...' : 'Connect with Existing Passkey'}
        </button>

        <button
          onClick={handleCreateNew}
          disabled={isConnecting || isPending}
          className="w-full px-4 py-3 border border-blue-500 text-blue-500 rounded hover:bg-blue-50 disabled:opacity-50"
        >
          {isConnecting || isPending ? 'Creating...' : 'Create New Wallet'}
        </button>
      </div>

      <p className="text-xs text-gray-500 mt-3 text-center">
        üîë Powered by WebAuthn passkeys
      </p>
    </div>
  )
}
```

## Step 9: Create Message Signing Component

Create `src/components/SignMessage.tsx`:

```tsx
// src/components/SignMessage.tsx
import { useState } from 'react'
import { useSignMessage, useAccount } from 'wagmi'

export function SignMessage() {
  const { address } = useAccount()
  const { signMessage, isPending, error } = useSignMessage()
  const [message, setMessage] = useState('Hello, HumanWallet!')
  const [signature, setSignature] = useState<string | null>(null)

  const handleSign = async () => {
    if (!message.trim()) return

    try {
      const sig = await signMessage({ message })
      setSignature(sig)
    } catch (err) {
      console.error('Failed to sign message:', err)
    }
  }

  if (!address) {
    return (
      <div className="p-6 border rounded-lg opacity-50">
        <h3 className="text-lg font-semibold mb-2">‚úçÔ∏è Sign Message</h3>
        <p className="text-sm text-gray-600">Connect your wallet first</p>
      </div>
    )
  }

  return (
    <div className="p-6 border rounded-lg">
      <h3 className="text-lg font-semibold mb-4">‚úçÔ∏è Sign Message</h3>
      
      {error && (
        <div className="mb-4 p-3 bg-red-50 border border-red-200 rounded">
          <p className="text-red-700 text-sm">{error.message}</p>
        </div>
      )}

      <div className="space-y-4">
        <div>
          <label className="block text-sm font-medium mb-2">
            Message to sign:
          </label>
          <textarea
            value={message}
            onChange={(e) => setMessage(e.target.value)}
            className="w-full p-3 border rounded resize-none"
            rows={3}
            placeholder="Enter your message..."
          />
        </div>

        <button
          onClick={handleSign}
          disabled={isPending || !message.trim()}
          className="px-4 py-2 bg-green-500 text-white rounded hover:bg-green-600 disabled:opacity-50"
        >
          {isPending ? 'Signing...' : 'Sign with Passkey'}
        </button>

        {signature && (
          <div className="p-3 bg-green-50 border border-green-200 rounded">
            <p className="text-sm font-medium text-green-700 mb-1">‚úÖ Signature:</p>
            <p className="text-xs text-green-600 font-mono break-all">
              {signature}
            </p>
          </div>
        )}
      </div>
    </div>
  )
}
```

## Common Issues & Solutions

### 1. "Project ID not found"
- Verify your ZeroDev project ID is correct
- Check that your environment variable is properly set
- Ensure you're not using a test project ID in production

### 2. "WebAuthn not supported"
- Make sure you're using HTTPS (or localhost for development)
- Verify the browser supports WebAuthn
- Check if the device has biometric capabilities

### 3. "Failed to create passkey"
- Ensure the domain is properly configured in ZeroDev
- Check browser settings for WebAuthn permissions
- Verify the user didn't cancel the authentication prompt
