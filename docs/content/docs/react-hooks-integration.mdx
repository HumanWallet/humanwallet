# React Hooks Integration

This comprehensive guide walks you through integrating the HumanWallet connector into your React application using Wagmi hooks. We'll build a complete integration that enables biometric authentication and account abstraction for your users.

## Prerequisites

Before we begin, make sure you have:

- **Node.js 18+** installed
- A **ZeroDev Project ID** (get one at [dashboard.zerodev.app](https://dashboard.zerodev.app/))
- Basic knowledge of **React** and **TypeScript**
- Familiarity with **React Hooks** and **Wagmi**

## Step 1: Create Your Project

First, create a new React project with Vite:

```bash
npm create vite@latest my-humanwallet-app -- --template react-ts
cd my-humanwallet-app
npm install
```

Install the required packages for HumanWallet integration:

## Step 6: Create Wagmi Provider

Create `src/wagmi/provider.tsx`:

```tsx
// src/wagmi/provider.tsx
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { WagmiProvider } from 'wagmi'
import { config } from './config'

const queryClient = new QueryClient()

export function WagmiProviders({ children }: { children: React.ReactNode }) {
  return (
    <WagmiProvider config={config}>
      <QueryClientProvider client={queryClient}>
        {children}
      </QueryClientProvider>
    </WagmiProvider>
  )
}
```

```tsx
// src/components/ConnectWallet.tsx
import { useConnect, useAccount, useDisconnect } from 'wagmi'
import { useState } from 'react'

export function ConnectWallet() {
  const { connectors, connect, isPending, error } = useConnect()
  const { address, isConnected } = useAccount()
  const { disconnect } = useDisconnect()
  const [isConnecting, setIsConnecting] = useState(false)

  // Find the HumanWallet connector
  const humanWallet = connectors.find(c => c.id === 'humanWallet')

  const handleConnect = async () => {
    if (!humanWallet) return
    setIsConnecting(true)
    try {
      await connect({ connector: humanWallet })
    } finally {
      setIsConnecting(false)
    }
  }

  const handleCreateNew = async () => {
    if (!humanWallet) return
    setIsConnecting(true)
    try {
      // Force create new wallet - this will prompt for a wallet name
      await (connect as any)({ 
        connector: humanWallet, 
        forceCreate: true 
      })
    } finally {
      setIsConnecting(false)
    }
  }

  const handleConnectWithUsername = async (username: string) => {
    if (!humanWallet) return
    setIsConnecting(true)
    try {
      // Create wallet with specific username
      await (connect as any)({ 
        connector: humanWallet, 
        username 
      })
    } finally {
      setIsConnecting(false)
    }
  }

  if (isConnected && address) {
    return (
      <div className="p-6 border rounded-lg bg-green-50 border-green-200">
        <h3 className="text-lg font-semibold mb-2 text-green-800">‚úÖ Connected</h3>
        <p className="text-sm text-green-700 mb-4">
          Address: <code className="bg-green-100 px-2 py-1 rounded text-xs">
            {address.slice(0, 6)}...{address.slice(-4)}

          onClick={() => disconnect()}
          className="px-4 py-2 bg-red-500 text-white rounded hover:bg-red-600 transition-colors"
        >
          Disconnect

    )
  }

  return (
    <div className="p-6 border rounded-lg">
      <h3 className="text-lg font-semibold mb-4">üîê Connect HumanWallet</h3>
      
      {error && (
        <div className="mb-4 p-3 bg-red-50 border border-red-200 rounded">
          <p className="text-red-700 text-sm font-medium">Connection Error</p>
          <p className="text-red-600 text-sm">{error.message}</p>
        </div>
      )}

      <div className="space-y-3">
        <button
          onClick={handleConnect}
          disabled={isConnecting || isPending}
          className="w-full px-4 py-3 bg-blue-500 text-white rounded hover:bg-blue-600 disabled:opacity-50 disabled:cursor-not-allowed transition-colors"
```

```tsx
// src/components/SignMessage.tsx
import { useState } from 'react'
import { useSignMessage, useAccount } from 'wagmi'

export function SignMessage() {
  const { address } = useAccount()
  const { signMessage, isPending, error } = useSignMessage()
  const [message, setMessage] = useState('Hello, HumanWallet!')
  const [signature, setSignature] = useState<string | null>(null)
  const [isVerified, setIsVerified] = useState(false)

  const handleSign = async () => {
    if (!message.trim()) return

    try {
      const sig = await signMessage({ message })
      setSignature(sig)
      setIsVerified(true)
    } catch (err) {
      console.error('Failed to sign message:', err)
      setIsVerified(false)
    }
  }

  const handleReset = () => {
    setSignature(null)
    setIsVerified(false)
  }

  if (!address) {
    return (
      <div className="p-6 border rounded-lg opacity-50 bg-gray-50">
        <h3 className="text-lg font-semibold mb-2 text-gray-600">‚úçÔ∏è Sign Message</h3>
        <p className="text-sm text-gray-500">Connect your wallet first to sign messages</p>
      </div>
    )
  }

  return (
    <div className="p-6 border rounded-lg">
      <h3 className="text-lg font-semibold mb-4">‚úçÔ∏è Sign Message with Passkey</h3>
      
      {error && (
        <div className="mb-4 p-3 bg-red-50 border border-red-200 rounded">
          <p className="text-red-700 text-sm font-medium">Signing Error</p>
          <p className="text-red-600 text-sm">{error.message}</p>
        </div>
      )}

      <div className="space-y-4">
        <div>
          <label className="block text-sm font-medium mb-2 text-gray-700">
            Message to sign:
          </label>
          <textarea
            value={message}
            onChange={(e) => setMessage(e.target.value)}
            className="w-full p-3 border border-gray-300 rounded-md resize-none focus:ring-2 focus:ring-blue-500 focus:border-blue-500 transition-colors"
            rows={3}
            placeholder="Enter your message..."
            disabled={isPending}
          />
        </div>

        <div className="flex gap-2">
          <button
            onClick={handleSign}

            className="flex-1 px-4 py-2 bg-green-500 text-white rounded hover:bg-green-600 disabled:opacity-50 disabled:cursor-not-allowed transition-colors"

            {isPending ? (
              <span className="flex items-center justify-center gap-2">
                <span className="w-4 h-4 border-2 border-white border-t-transparent rounded-full animate-spin"></span>
                Signing...
              </span>
            ) : (
              'üîê Sign with Passkey'
            )}
          </button>

          {signature && (
            <button
              onClick={handleReset}
              className="px-4 py-2 border border-gray-300 text-gray-700 rounded hover:bg-gray-50 transition-colors"
            >
              Reset
            </button>
          )}
        </div>

        {signature && (
          <div className="p-4 bg-green-50 border border-green-200 rounded-lg">
            <div className="flex items-center gap-2 mb-2">
              <span className="text-green-600">‚úÖ</span>
              <p className="text-sm font-medium text-green-700">
                Message signed successfully!
              </p>
            </div>
```

```tsx
// src/components/SendTransaction.tsx
import { useState } from 'react'
import { useSendTransaction, useAccount, useWaitForTransactionReceipt } from 'wagmi'
import { parseEther, isAddress } from 'viem'

export function SendTransaction() {
  const { address } = useAccount()
  const { sendTransaction, isPending, error } = useSendTransaction()
  const [to, setTo] = useState('')
  const [amount, setAmount] = useState('')
  const [txHash, setTxHash] = useState<`0x${string}` | null>(null)

  // Watch for transaction confirmation
  const { isLoading: isConfirming, isSuccess } = useWaitForTransactionReceipt({
    hash: txHash,
  })

  const handleSend = async () => {
    if (!to || !amount || !isAddress(to)) return

    try {
      const hash = await sendTransaction({
        to: to as `0x${string}`,
        value: parseEther(amount),
      })
      setTxHash(hash)
    } catch (err) {
      console.error('Failed to send transaction:', err)
    }
  }

  const handleReset = () => {
    setTo('')
    setAmount('')
    setTxHash(null)
  }

  if (!address) {
    return (
      <div className="p-6 border rounded-lg opacity-50 bg-gray-50">
        <h3 className="text-lg font-semibold mb-2 text-gray-600">üí∏ Send Transaction</h3>
        <p className="text-sm text-gray-500">Connect your wallet first to send transactions</p>
      </div>
    )
  }

  return (
    <div className="p-6 border rounded-lg">
      <h3 className="text-lg font-semibold mb-4">üí∏ Send Transaction</h3>
      
      {error && (
        <div className="mb-4 p-3 bg-red-50 border border-red-200 rounded">
          <p className="text-red-700 text-sm font-medium">Transaction Error</p>
          <p className="text-red-600 text-sm">{error.message}</p>

      {isSuccess && txHash && (
        <div className="mb-4 p-3 bg-green-50 border border-green-200 rounded">
          <p className="text-green-700 text-sm font-medium">‚úÖ Transaction Confirmed!</p>
          <p className="text-green-600 text-xs mt-1 break-all">
            Hash: {txHash}
          </p>
        </div>
      )}

      <div className="space-y-4">
        <div>
          <label className="block text-sm font-medium mb-2 text-gray-700">
            To Address:
          </label>
          <input
            type="text"
            value={to}
            onChange={(e) => setTo(e.target.value)}
            className="w-full p-3 border border-gray-300 rounded-md focus:ring-2 focus:ring-blue-500 focus:border-blue-500 transition-colors"

          />
        </div>

        <div className="flex gap-2">
          <button
            onClick={handleSend}
            disabled={isPending || isConfirming || !to || !amount || !isAddress(to)}

            {isPending ? (
```

## Advanced Hook Patterns

### Custom Hook for HumanWallet

Create `src/hooks/useHumanWallet.ts`:

```typescript
// src/hooks/useHumanWallet.ts
import { useConnect, useAccount, useDisconnect } from 'wagmi'
import { useState } from 'react'

export function useHumanWallet() {
  const { connectors, connect, isPending, error } = useConnect()
  const { address, isConnected } = useAccount()
  const { disconnect } = useDisconnect()
  const [isConnecting, setIsConnecting] = useState(false)

  const humanWallet = connectors.find(c => c.id === 'humanWallet')

  const connectExisting = async () => {
    if (!humanWallet) return
    setIsConnecting(true)
    try {
      await connect({ connector: humanWallet })
    } finally {
      setIsConnecting(false)
    }
  }

  const createNew = async (username?: string) => {
    if (!humanWallet) return
    setIsConnecting(true)
    try {
      await (connect as any)({ 
        connector: humanWallet, 
        ...(username ? { username } : { forceCreate: true })
      })
    } finally {
      setIsConnecting(false)
    }
  }

  return {
    // State
    address,
    isConnected,
    isConnecting: isConnecting || isPending,
    error,
    
    // Actions
    connectExisting,
    createNew,
    disconnect,
    
    // Connector info
    isAvailable: !!humanWallet,
  }
}
```

```tsx
import { useHumanWallet } from '../hooks/useHumanWallet'

export function SimpleConnect() {
  const { 
    address, 
    isConnected, 
    isConnecting, 
    connectExisting, 
    createNew, 
    disconnect 
  } = useHumanWallet()

  if (isConnected) {
    return (
      <div>
        <p>Connected: {address}</p>
        <button onClick={() => disconnect()}>Disconnect</button>
      </div>
    )
  }

  return (
    <div>
      <button 
        onClick={connectExisting} 
        disabled={isConnecting}
      >
        Connect Existing
      </button>
      <button 
        onClick={() => createNew()} 
        disabled={isConnecting}
      >
        Create New
      </button>
    </div>
  )
}
```
